Paste this into the browser console on the zWallet page:

// Check what's blocking the tabs
const tabs = document.querySelectorAll('.tab');
tabs.forEach((tab) => {
  const rect = tab.getBoundingClientRect();
  const el = document.elementFromPoint(rect.left + rect.width/2, rect.top + rect.height/2);
  if (el !== tab) {
    console.warn(`Tab ${tab.dataset.tab} is BLOCKED by:`, el);
  } else {
    console.log(`Tab ${tab.dataset.tab} is clickable`);
  }
});
